/*global define*/
define(['angular', 'jquery', 'underscore', 'url'], function (angular, $, _, url) {
    'use strict';
    var paymentController = function ($scope, $timeout, $http) {
 
        $scope.checkPay = function () {
            return !$scope.payment.amount || !$scope.payment.payMethod || !$scope.payment.tendered;
        };
 
        $scope.rateDisplay = function () {
            if ($scope.payment.rate || $scope.payment.tendered) {
                return 'x ' + $scope.payment.rate + ' = ' + $scope.payment.rate * $scope.payment.tendered;
            }
        };
 
        $scope.calChange = function () {
            if ($scope.payment.tendered) {
                var tendered = $scope.payment.rate ? $scope.payment.rate * $scope.payment.tendered : $scope.payment.tendered;
                var change = ($scope.payment.amount - tendered) * -1;
                return change < 0 ? 0 : change;
            }
        };
 
        var loadExchangeRate = function () {
            $scope.clear();
            if ($scope.payment.payMethod >= 100) {
                $http.get(url.resolve('/Service/Payments/GetExchangeRate?payMethod=') + $scope.payment.payMethod)
                    .success(function (data) {
                        $scope.payment.rate = parseFloat(data);
                    }).error(function () {
                        alert('Could not load exchange rate.', 'Load Failed');
                    });
            } else {
                $scope.payment.rate = null;
            }
        };
 
        $scope.choosePaymentMethod = function () {
            $scope.payment.payMethod = this.value + '';
            loadExchangeRate();
        };
 
        $scope.savePayment = function () {
            var tendered = $scope.payment.rate ? $scope.payment.rate * parseFloat($scope.payment.tendered, 10) : parseFloat($scope.payment.tendered, 10);
            $scope.cart.payments.push({
                method: $scope.MasterData.payMethods[$scope.payment.payMethod],
                amount: tendered
            });
            $scope.cart.changeTotal += $scope.calChange();
            $scope.cart.paymentsTotal += tendered;
            $scope.payment.tendered = null;
            $scope.updateBalance();
            $scope.cart.basketChangesAllowed = false;
 
            if ($scope.sections.change.visible) {
                $scope.$emit('shell:till:open');
            }
 
            if ($scope.cart.balance === 0) {
                $scope.$emit('pos:sale:completed', $scope.cart);
            }
        };
 
        $scope.restrictTenderedAmount = function () {
            if ($scope.sections.change.visible) {
                return;
            }
 
            if ($scope.payment.tendered > $scope.payment.amount) {
                $scope.payment.tendered = $scope.payment.amount;
            }
        };
 
        $scope.clear = function () {
            delete $scope.payment.bankAccountNumber;
            delete $scope.payment.bank;
            delete $scope.payment.chequeNumber;
            delete $scope.payment.cardType;
            delete $scope.payment.cardNumber;
        };
    };
 
    return paymentController;
});
